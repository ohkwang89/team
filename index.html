const SHEET_NAME = '시트1'; function formatDate(val) { if (!val) return ''; if (val instanceof Date) { const y = val.getFullYear(); const m = String(val.getMonth() + 1).padStart(2, '0'); const d = String(val.getDate()).padStart(2, '0'); return y + '-' + m + '-' + d; } const s = String(val).trim(); if (!s) return ''; if (s.includes('T')) return s.slice(0, 10); if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; const parsed = new Date(s); if (!isNaN(parsed.getTime())) { const y = parsed.getFullYear(); const mo = String(parsed.getMonth() + 1).padStart(2, '0'); const d = String(parsed.getDate()).padStart(2, '0'); return y + '-' + mo + '-' + d; } return s; } function formatTime(val) { if (!val) return ''; if (val instanceof Date) { const h = String(val.getHours()).padStart(2, '0'); const m = String(val.getMinutes()).padStart(2, '0'); return h + ':' + m; } const s = String(val).trim(); // HH:MM 형식이면 그대로 if (/^\d{2}:\d{2}$/.test(s)) return s; // 소수점 형식(0.375 같은 시트 시간값)이면 변환 const n = parseFloat(s); if (!isNaN(n) && n >= 0 && n < 1) { const totalMin = Math.round(n * 24 * 60); const h = String(Math.floor(totalMin / 60)).padStart(2, '0'); const m = String(totalMin % 60).padStart(2, '0'); return h + ':' + m; } return s; } function setTextCell(sheet, row, col, value) { if (col < 0) return; const cell = sheet.getRange(row, col + 1); cell.setNumberFormat('@'); cell.setValue(value); } function doGet(e) { const callback = e.parameter.callback; const action = e.parameter.action; const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName(SHEET_NAME); let result; if (action === 'save') { const data = sheet.getDataRange().getValues(); const headers = data[0]; const rows = data.slice(1); const idCol = headers.indexOf('id'); const startCol = headers.indexOf('start'); const endCol = headers.indexOf('end'); const stimeCol = headers.indexOf('stime'); const etimeCol = headers.indexOf('etime'); const incomingId = String(e.parameter.id).trim(); const newRow = headers.map(h => { const v = e.parameter[h] || ''; if (h === 'start' || h === 'end') return formatDate(v); if (h === 'stime' || h === 'etime') return v; // 일단 텍스트로 return v; }); let targetRow; const existingIdx = rows.findIndex(r => String(r[idCol]).trim() === incomingId); if (existingIdx >= 0) { targetRow = existingIdx + 2; sheet.getRange(targetRow, 1, 1, newRow.length).setValues([newRow]); } else { sheet.appendRow(newRow); targetRow = sheet.getLastRow(); } // start, end, stime, etime 모두 텍스트 서식으로 강제 지정 setTextCell(sheet, targetRow, startCol, formatDate(e.parameter.start || '')); setTextCell(sheet, targetRow, endCol, formatDate(e.parameter.end || '')); setTextCell(sheet, targetRow, stimeCol, e.parameter.stime || ''); setTextCell(sheet, targetRow, etimeCol, e.parameter.etime || ''); result = {result: 'ok'}; } else if (action === 'delete') { const data = sheet.getDataRange().getValues(); const idCol = data[0].indexOf('id'); const targetId = String(e.parameter.id).trim(); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][idCol]).trim() === targetId) { sheet.deleteRow(i + 1); break; } } result = {result: 'ok'}; } else { const data = sheet.getDataRange().getValues(); const headers = data[0]; result = data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { let v = row[i]; if (h === 'start' || h === 'end') v = formatDate(v); else if (h === 'stime' || h === 'etime') v = formatTime(v); else v = (v === null || v === undefined) ? '' : String(v); obj[h] = v; }); return obj; }).filter(row => String(row.id).trim() !== ''); } const json = JSON.stringify(result); const output = callback ? ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT) : ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); return output; }
